# Pipeline V2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the 15-step legal assistant pipeline with a 5-step per-issue retrieval architecture that eliminates the reranker, graph expansion, and Step 12 re-runs — cutting cost from $0.40 to ~$0.15, latency from 255s to ~100s, and improving legal accuracy.

**Architecture:** New `pipeline_v2_service.py` implements 5 steps: Classify -> Resolve+Gate -> Retrieve-per-issue -> RL-RAP (once) -> Answer+Validate. Reuses existing `chroma_service.py`, `bm25_service.py`, `claude_service.py`, `law_mapping.py`, `version_currency.py` unchanged. The old `pipeline_service.py` is preserved as fallback.

**Tech Stack:** Python 3.12, FastAPI, SQLAlchemy, ChromaDB, SQLite FTS5, Claude API

**Spec:** `docs/superpowers/specs/2026-04-02-pipeline-v2-design.md`

def step1_classify(state: dict, db: Session) -> dict:

    Populates: legal_issues, applicable_laws, entity_types, facts, complexity,
               primary_target, output_mode, legal_domain, etc.

# Step 2A: Version Selection Per Fact
# ---------------------------------------------------------------------------

def step2a_version_selection(state: dict, db: Session) -> dict:
    """Select the correct law version for each (issue, law) pair based on temporal rules.
    
    Populates: fact_version_map, unique_versions
    """
    fact_version_map = {}
    unique_versions = {}  # law_key -> set of version_ids

    # Build law_id lookup from candidate_laws
    law_id_lookup = {}  # law_key -> db_law_id
    for cl in state.get("candidate_laws", []):
        key = f"{cl['law_number']}/{cl['law_year']}"
        if cl.get("db_law_id"):
            law_id_lookup[key] = cl["db_law_id"]

    for issue in state.get("legal_issues", []):
        issue_id = issue["issue_id"]

        for law_key in issue.get("applicable_laws", []):
            db_law_id = law_id_lookup.get(law_key)
            if not db_law_id:
                continue

            # Determine the reference date for this issue+law
            # First check fact_dates for per-fact dates
            reference_date = issue.get("relevant_date", state["today"])
            fact_dates = issue.get("fact_dates", [])
            law_fact_dates = [fd for fd in fact_dates if law_key in fd.get("applicable_laws", [])]

            if law_fact_dates:
                # Use the latest fact date for this law
                reference_date = max(fd["relevant_date"] for fd in law_fact_dates)

            # Query versions for this law, ordered by date_in_force DESC
            versions = db.execute(
                select(LawVersion)
                .where(LawVersion.law_id == db_law_id)
                .order_by(LawVersion.date_in_force.desc())
            ).scalars().all()

            # Select: latest version where date_in_force <= reference_date
            selected = None
            for v in versions:
                v_date = str(v.date_in_force) if v.date_in_force else ""
                if v_date and v_date <= reference_date:
                    selected = v
                    break

            map_key = f"{issue_id}:{law_key}"
            if selected:
                fact_version_map[map_key] = {
                    "law_version_id": selected.id,
                    "date_in_force": str(selected.date_in_force),
                    "reference_date": reference_date,
                }
                unique_versions.setdefault(law_key, set()).add(selected.id)
            else:
                # No valid version found — try fallback to latest available
                if versions:
                    fallback = versions[0]
                    fact_version_map[map_key] = {
                        "law_version_id": fallback.id,
                        "date_in_force": str(fallback.date_in_force),
                        "reference_date": reference_date,
                        "fallback": True,
                    }
                    unique_versions.setdefault(law_key, set()).add(fallback.id)
                    state.setdefault("flags", []).append(
                        f"Version fallback: {law_key} has no version for {reference_date}, using latest"
                    )
                else:
                    fact_version_map[map_key] = {
                        "law_version_id": None,
                        "reference_date": reference_date,
                    }

            # Check mitior_lex: if criminal issue, flag newer version
            if issue.get("mitior_lex_relevant") and selected:
                newer = [v for v in versions if str(v.date_in_force) > reference_date]
                if newer:
                    state.setdefault("flags", []).append(
                        f"Mitior lex: {law_key} has newer version ({newer[0].date_in_force}) "
                        f"than act date ({reference_date}). Art. 5 Cod Penal may apply."
                    )

    state["fact_version_map"] = fact_version_map
    state["unique_versions"] = {k: list(v) for k, v in unique_versions.items()}
    return state

    """Search ChromaDB per issue using entity-aware concept descriptions.
    
    Populates: concept_candidates (dict: issue_id -> list of article dicts)
    """
    concept_candidates = {}
    fvm = state.get("fact_version_map", {})

    for issue in state.get("legal_issues", []):
        issue_id = issue["issue_id"]
        candidates = []
        seen_ids = set()

        for concept in issue.get("concept_descriptions", []):
            law_key = concept.get("law_key", "")
            concept_text = concept.get("concept_general", "")
            if not concept_text or not law_key:
                continue

            # Get version ID for this issue+law
            map_key = f"{issue_id}:{law_key}"
            version_info = fvm.get(map_key, {})
            vid = version_info.get("law_version_id")
            if not vid:
                continue

            # Semantic search within this specific version
            try:
                results = query_articles(
                    query_text=concept_text,
                    law_version_ids=[vid],
                    n_results=N_SEMANTIC,
                )
                for r in results:
                    aid = r.get("article_id")
                    if aid in seen_ids:
                        continue
                    # Filter abrogated
                    if str(r.get("is_abrogated", "False")).lower() == "true":
                        continue
                    seen_ids.add(aid)
                    r["source"] = "concept_search"
                    r["law_key"] = law_key
                    r["issue_id"] = issue_id
                    candidates.append(r)
            except Exception as e:
                logger.warning(f"Concept search failed for {issue_id}/{law_key}: {e}")

        concept_candidates[issue_id] = candidates

    state["concept_candidates"] = concept_candidates
    return state

def step2e_availability_gate(state: dict, db: Session) -> dict | None:
    """Check if all required laws and versions are available.
    
    Returns None if pipeline should continue.
    Returns a gate event dict if pipeline should pause or stop.
    """
    from app.services.pipeline_logger import save_paused_state

    candidate_laws = state.get("candidate_laws", [])
    if not candidate_laws:
        return {
            "type": "done",
            "mode": "clarification",
            "content": "Nu am putut identifica legi aplicabile pentru această întrebare. "
                       "Vă rugăm să reformulați sau să oferiți mai multe detalii.",
        }

    primary_laws = [c for c in candidate_laws if c.get("role") == "PRIMARY"]
    missing = [c for c in primary_laws if c.get("availability") in ("missing", "wrong_version")]
    stale = [c for c in primary_laws if c.get("currency_status") == "stale"]

    if missing or stale:
        # Build import options for the frontend
        laws_for_import = []
        for c in missing + stale:
            laws_for_import.append({
                "law_key": f"{c['law_number']}/{c['law_year']}",
                "title": c.get("title", ""),
                "status": c.get("availability", c.get("currency_status", "unknown")),
                "available_version_date": c.get("available_version_date"),
                "official_latest_date": c.get("official_latest_date"),
            })

        # Save state for resume
        save_paused_state(db, state["run_id"], state)
        db.commit()

        return {
            "type": "pause",
            "run_id": state["run_id"],
            "laws": laws_for_import,
        }

    return None

def step3_retrieve_per_issue(state: dict, db: Session) -> dict:
    """Retrieve articles separately for each issue, scoped to correct versions.
    
    Populates: issue_articles (dict: issue_id -> list of article dicts)
    """
    issue_articles = {}
    concept_candidates = state.get("concept_candidates", {})
    unique_versions = state.get("unique_versions", {})
    fvm = state.get("fact_version_map", {})

    for issue in state.get("legal_issues", []):
        issue_id = issue["issue_id"]
        priority = issue.get("priority", "SECONDARY")
        budget = 12 if priority == "PRIMARY" else (8 if priority == "SECONDARY" else 5)

        articles = {}  # article_id -> article_dict (dedup)

        # 1. Start with concept candidates from Step 2B
        for art in concept_candidates.get(issue_id, []):
            aid = art.get("article_id")
            if aid:
                articles[aid] = art

        # 2. BM25 search per law, scoped to issue's versions
        for law_key in issue.get("applicable_laws", []):
            map_key = f"{issue_id}:{law_key}"
            version_info = fvm.get(map_key, {})
            vid = version_info.get("law_version_id")
            if not vid:
                continue

            # Build BM25 query from issue description + entity terms
            bm25_query = issue.get("description", "")
            entity = issue.get("entity_perspective", {})
            if isinstance(entity, dict):
                actor = entity.get("actor", "")
                if actor:
                    bm25_query = f"{actor} {bm25_query}"

            try:
                bm25_results = search_bm25(
                    db=db,
                    query=bm25_query,
                    law_version_ids=[vid],
                    limit=N_BM25,
                )
                for r in bm25_results:
                    aid = r.get("article_id")
                    if aid and aid not in articles:
                        r["source"] = "bm25"
                        r["law_key"] = law_key
                        r["issue_id"] = issue_id
                        articles[aid] = r
            except Exception as e:
                logger.warning(f"BM25 failed for {issue_id}/{law_key}: {e}")

        # 3. Additional semantic search with issue description
        for law_key in issue.get("applicable_laws", []):
            map_key = f"{issue_id}:{law_key}"
            version_info = fvm.get(map_key, {})
            vid = version_info.get("law_version_id")
            if not vid:
                continue

            try:
                sem_results = query_articles(
                    query_text=issue.get("description", state["question"]),
                    law_version_ids=[vid],
                    n_results=N_SEMANTIC,
                )
                for r in sem_results:
                    aid = r.get("article_id")
                    if aid and aid not in articles:
                        if str(r.get("is_abrogated", "False")).lower() == "true":
                            continue
                        r["source"] = "semantic"
                        r["law_key"] = law_key
                        r["issue_id"] = issue_id
                        articles[aid] = r
            except Exception as e:
                logger.warning(f"Semantic search failed for {issue_id}/{law_key}: {e}")

        # Trim to budget (prefer concept candidates, then by distance/rank)
        article_list = list(articles.values())
        if len(article_list) > budget:
            # Sort: concept_search first, then by distance (lower is better)
            def sort_key(a):
                source_priority = 0 if a.get("source") == "concept_search" else 1
                dist = a.get("distance", 999)
                return (source_priority, dist)
            article_list.sort(key=sort_key)
            article_list = article_list[:budget]

        issue_articles[issue_id] = article_list

    state["issue_articles"] = issue_articles

    # Compute lightweight relevance signal
    max_distances = []
    for issue in state.get("legal_issues", []):
        if issue.get("priority") == "PRIMARY":
            arts = issue_articles.get(issue["issue_id"], [])
            dists = [a.get("distance", 999) for a in arts if a.get("distance") is not None]
            if dists:
                max_distances.append(min(dists))
    if max_distances and min(max_distances) > 0.7:
        state.setdefault("flags", []).append("low_relevance_signal")

    return state

def step4_legal_reasoning(state: dict, db: Session) -> dict:
    """Structured legal reasoning using RL-RAP methodology.
    
    Runs exactly once. No conditional retrieval loop.
    Populates: rl_rap_output, operative_articles
    """
    import json
    from app.services.pipeline_service import _extract_json

    # Dynamic token budget
    num_issues = len(state.get("legal_issues", []))
    if state.get("complexity") == "COMPLEX" or num_issues >= 3:
        max_tokens = min(16384, 4096 + num_issues * 2048)
    else:
        max_tokens = 8192

    prompt_text = load_prompt(db, "LA-V2-S4-reasoning")
    if not prompt_text:
        import os
        prompt_path = os.path.join(os.path.dirname(__file__), "..", "..", "prompts", "LA-V2-S4-reasoning.txt")
        with open(prompt_path) as f:
            prompt_text = f.read()

    # Build context from per-issue articles
    from app.services.pipeline_v2_context import build_step4_context
    context = build_step4_context(state)

    response = call_claude(
        system=prompt_text,
        user_message=context,
        max_tokens=max_tokens,
        temperature=0.1,
        step_name="v2_step4_reasoning",
        run_id=state.get("run_id"),
        db=db,
    )

    raw_text = response.get("text", "")
    stop_reason = response.get("stop_reason", "")

    # Check for truncation
    if stop_reason == "max_tokens":
        logger.warning(f"Step 4 truncated at {max_tokens} tokens. Response may be incomplete.")
        state.setdefault("flags", []).append(f"rl_rap_truncated_at_{max_tokens}")

    parsed = _extract_json(raw_text)
    if parsed and "issues" in parsed:
        state["rl_rap_output"] = parsed
        # Extract operative articles
        operative = []
        for issue in parsed.get("issues", []):
            for oa in issue.get("operative_articles", []):
                operative.append(oa.get("article_ref", ""))
        state["operative_articles"] = operative
    else:
        logger.error(f"Step 4 failed to parse RL-RAP output. Raw ({len(raw_text)} chars): {raw_text[:500]}")
        state["rl_rap_output"] = None
        state["operative_articles"] = []
        state.setdefault("flags", []).append("rl_rap_parse_failed")

    return state

def step5_answer_generation(state: dict, db: Session):

    Yields streaming token events. Updates state with answer and structured output.
    """
    import json

        import os
        prompt_path = os.path.join(os.path.dirname(__file__), "..", "..", "prompts", "LA-V2-S5-answer.txt")
        with open(prompt_path) as f:
            prompt_text = f.read()

    # Load mode-specific prompt
    mode = state.get("output_mode", "qa")
    if state.get("complexity") == "SIMPLE" and not state.get("rl_rap_output"):
        mode = "simple"
    mode_prompt = load_prompt(db, f"LA-S7-mode-{mode}")
    if not mode_prompt:
        import os
        mode_path = os.path.join(os.path.dirname(__file__), "..", "..", "prompts", f"LA-S7-mode-{mode}.txt")
        try:
            with open(mode_path) as f:
                mode_prompt = f.read()
        except FileNotFoundError:
            mode_prompt = ""

    system_msg = prompt_text + "\n\n" + mode_prompt

    # Build context
    from app.services.pipeline_v2_context import build_step5_context
    context = build_step5_context(state)

    # Stream the response
    full_text = ""
    for chunk in stream_claude(
        system=system_msg,
        user_message=context,
        max_tokens=8192,
        temperature=0.2,
        step_name="v2_step5_answer",
        run_id=state.get("run_id"),
        db=db,
    ):
        if isinstance(chunk, dict) and chunk.get("type") == "text":
            full_text += chunk["text"]
            yield {"type": "token", "text": chunk["text"]}
        elif isinstance(chunk, str):
            full_text += chunk
            yield {"type": "token", "text": chunk}

    state["answer"] = full_text

    # Try to parse structured answer
    parsed = _extract_json(full_text)
    if parsed:
        state["answer_structured"] = parsed
        state["claude_confidence"] = parsed.get("confidence", "MEDIUM")
    else:
        state["answer_structured"] = {"answer": full_text}
        state["claude_confidence"] = "MEDIUM"


def step5b_citation_validation(state: dict, db: Session) -> dict:
    """Validate citations in the structured answer against retrieved articles.
    
    Downgrades phantom citations from "DB" to "Unverified".
    """
    structured = state.get("answer_structured", {})
    sources = structured.get("sources", [])
    if not sources:
        return state

    # Build set of article IDs that were actually provided
    provided_articles = set()
    for issue_id, arts in state.get("issue_articles", {}).items():
        for a in arts:
            art_ref = f"{a.get('law_number', '')}/{a.get('law_year', '')} art.{a.get('article_number', '')}"
            provided_articles.add(art_ref)
            # Also add without spaces for fuzzy matching
            provided_articles.add(art_ref.replace(" ", ""))

    phantom_count = 0
    total_db = 0
    for source in sources:
        if source.get("source_type") == "DB":
            total_db += 1
            ref = source.get("article_ref", "")
            # Check if this citation was in the provided articles
            normalized = ref.replace(" ", "").replace("Art.", "art.").replace("art.", "art.")
            found = any(normalized in pa or pa in normalized for pa in provided_articles)
            if not found:
                source["source_type"] = "Unverified"
                source["validation_note"] = "Citation not found in provided articles"
                phantom_count += 1

    if total_db > 0 and phantom_count > total_db / 2:
        state.setdefault("flags", []).append("majority_phantom_citations")

    state["phantom_citation_count"] = phantom_count
    return state

"""
Pipeline V2 — 5-step legal reasoning engine.

Steps:
  1. Classify + Temporal Anchor (Claude)
  2. Resolve + Version + Gate (local)
  3. Retrieve Per-Issue (local)
  4. Legal Reasoning / RL-RAP (Claude, ONCE)
  5. Answer + Validate (Claude streaming + local)

SIMPLE path: 1 -> 2 -> 3 -> 5 (skip Step 4)
STANDARD/COMPLEX path: 1 -> 2 -> 3 -> 4 -> 5
"""
from __future__ import annotations

import datetime
import logging
import time
from typing import Generator

from sqlalchemy.orm import Session

from app.services.pipeline_logger import (
    complete_run,
    create_run,
    log_step,
    load_paused_state,
    save_paused_state,
    update_run_mode,
)
from app.services.pipeline_v2_steps import (
    step1_classify,
    step2a_version_selection,
    step2b_concept_search,
    step2c_law_availability,
    step2d_currency_check,
    step2e_availability_gate,
    step3_retrieve_per_issue,
    step4_legal_reasoning,
    step5_answer_generation,
    step5b_citation_validation,
)

logger = logging.getLogger(__name__)


def _step_event(step: int, name: str, status: str, data: dict | None = None, duration: float = 0) -> dict:
    return {"type": "step", "step": step, "name": name, "status": status, "data": data or {}, "duration": round(duration, 2)}


def run_pipeline_v2(
    question: str,
    session_context: list[dict],
    db: Session,
) -> Generator[dict, None, None]:
    """Run the V2 5-step pipeline. Yields SSE-compatible event dicts."""
    run_id = create_run(db, question)
    state = {
        "question": question,
        "session_context": session_context,
        "run_id": run_id,
        "flags": [],
        "today": datetime.date.today().isoformat(),
    }

    try:
        # ─── STEP 1: Classify + Temporal Anchor (LLM) ───
        yield _step_event(1, "classify", "running")
        t0 = time.time()
        state = step1_classify(state, db)
        dur = time.time() - t0
        log_step(db, run_id, "classify", 1, "done", dur,
                 output_summary=f"complexity={state.get('complexity')} mode={state.get('output_mode')} issues={len(state.get('legal_issues', []))}",
                 output_data={
                     "complexity": state.get("complexity"),
                     "output_mode": state.get("output_mode"),
                     "legal_domain": state.get("legal_domain"),
                     "entity_types": state.get("entity_types", []),
                     "legal_issues": [
                         {"id": i["issue_id"], "description": i.get("description", ""), "priority": i.get("priority", "")}
                         for i in state.get("legal_issues", [])
                     ],
                 })
        update_run_mode(db, run_id, state.get("output_mode"))
        yield _step_event(1, "classify", "done", {
            "mode": state.get("output_mode"),
            "domain": state.get("legal_domain"),
            "complexity": state.get("complexity"),
            "entity_types": state.get("entity_types", []),
            "issues_count": len(state.get("legal_issues", [])),
        }, dur)

        # ─── STEP 2: Resolve + Version + Gate (local) ───
        yield _step_event(2, "resolve", "running")
        t0 = time.time()

        # 2C: Law availability (must run before 2A so we have db_law_ids)
        state = step2c_law_availability(state, db)

        # 2A: Version selection per fact
        state = step2a_version_selection(state, db)

        # 2B: Entity-aware concept search per issue
        state = step2b_concept_search(state, db)

        # 2D: Version currency check (only for current-law issues)
        state = step2d_currency_check(state, db)

        dur = time.time() - t0
        n_concepts = sum(len(v) for v in state.get("concept_candidates", {}).values())
        log_step(db, run_id, "resolve", 2, "done", dur,
                 output_summary=f"versions={len(state.get('fact_version_map', {}))} concepts={n_concepts}",
                 output_data={
                     "fact_version_map": state.get("fact_version_map", {}),
                     "concept_candidates_per_issue": {
                         k: len(v) for k, v in state.get("concept_candidates", {}).items()
                     },
                     "coverage_status": state.get("coverage_status", {}),
                 })

        # 2E: Availability gate
        gate_result = step2e_availability_gate(state, db)
        if gate_result:
            log_step(db, run_id, "resolve", 2, "done", time.time() - t0,
                     output_summary=f"Gate triggered: {gate_result.get('type')}",
                     warnings=["Pipeline paused or stopped"])
            yield _step_event(2, "resolve", "done", {"gate_triggered": True}, time.time() - t0)
            if gate_result.get("type") == "pause":
                yield gate_result
                return
            else:
                complete_run(db, run_id, "clarification", None, state.get("flags"))
                db.commit()
                yield gate_result
                return

        yield _step_event(2, "resolve", "done", {
            "versions_resolved": len(state.get("fact_version_map", {})),
            "concept_candidates": n_concepts,
            "gate_triggered": False,
        }, dur)

        # ─── STEP 3: Retrieve Per-Issue (local) ───
        yield _step_event(3, "retrieve", "running")
        t0 = time.time()
        state = step3_retrieve_per_issue(state, db)
        dur = time.time() - t0
        total_articles = sum(len(v) for v in state.get("issue_articles", {}).values())
        log_step(db, run_id, "retrieve", 3, "done", dur,
                 output_summary=f"total_articles={total_articles}",
                 output_data={
                     "per_issue": {
                         k: [{"article_number": a.get("article_number"), "law": a.get("law_key"), "source": a.get("source")}
                              for a in v]
                         for k, v in state.get("issue_articles", {}).items()
                     },
                 })
        yield _step_event(3, "retrieve", "done", {
            "total_articles": total_articles,
            "per_issue": {k: len(v) for k, v in state.get("issue_articles", {}).items()},
        }, dur)

        # ─── STEP 4: Legal Reasoning / RL-RAP (LLM, ONCE) ───
        if state.get("complexity") != "SIMPLE":
            yield _step_event(4, "reasoning", "running")
            t0 = time.time()
            state = step4_legal_reasoning(state, db)
            dur = time.time() - t0

            rl_rap = state.get("rl_rap_output")
            rl_summary = "failed" if not rl_rap else f"issues={len(rl_rap.get('issues', []))}"
            log_step(db, run_id, "reasoning", 4, "done", dur,
                     output_summary=rl_summary,
                     output_data={
                         "rl_rap_success": rl_rap is not None,
                         "issues_analyzed": len(rl_rap.get("issues", [])) if rl_rap else 0,
                         "certainty_levels": {
                             i["issue_id"]: i.get("certainty_level", "UNKNOWN")
                             for i in (rl_rap or {}).get("issues", [])
                         },
                     },
                     warnings=["rl_rap_parse_failed"] if not rl_rap else None)
            yield _step_event(4, "reasoning", "done", {
                "success": rl_rap is not None,
                "issues_analyzed": len(rl_rap.get("issues", [])) if rl_rap else 0,
            }, dur)

        # ─── STEP 5: Answer + Validate (LLM streaming + local) ───
        yield _step_event(5, "answer", "running")
        t0 = time.time()
        for event in step5_answer_generation(state, db):
            yield event
        dur_answer = time.time() - t0

        # Citation validation
        state = step5b_citation_validation(state, db)

        # Derive final confidence
        confidence = _derive_confidence(state)
        state["confidence"] = confidence

        log_step(db, run_id, "answer", 5, "done", dur_answer,
                 output_summary=f"confidence={confidence}",
                 output_data={
                     "confidence": confidence,
                     "phantom_citations": state.get("phantom_citation_count", 0),
                 })
        yield _step_event(5, "answer", "done", {
            "confidence": confidence,
        }, dur_answer)

        # ─── FINALIZE ───
        complete_run(db, run_id, "ok", confidence, state.get("flags"))
        db.commit()

        yield {
            "type": "done",
            "run_id": run_id,
            "content": state.get("answer", ""),
            "structured": state.get("answer_structured"),
            "mode": state.get("output_mode", "qa"),
            "confidence": confidence,
            "flags": state.get("flags", []),
            "reasoning": _build_v2_reasoning_panel(state),
        }

    except GeneratorExit:
        logger.info(f"Pipeline V2 {run_id} interrupted by client disconnect")
    except (OSError, IOError) as e:
        logger.info(f"Pipeline V2 {run_id} connection lost: {e}")
    except Exception as e:
        logger.exception(f"Pipeline V2 error in run {run_id}")
        error_detail = str(e)
        if e.__cause__:
            error_detail += f" | Cause: {type(e.__cause__).__name__}: {e.__cause__}"
        try:
            complete_run(db, run_id, "error", None, [error_detail])
            db.commit()
        except Exception:
            pass
        yield {"type": "error", "error": str(e), "run_id": run_id}


def resume_pipeline_v2(
    run_id: str,
    import_decisions: dict[str, str],
    db: Session,
) -> Generator[dict, None, None]:
    """Resume V2 pipeline after import pause.
    
    Reloads saved state and re-runs from Step 2C onward.
    """
    saved = load_paused_state(db, run_id)
    if not saved:
        yield {"type": "error", "error": "No paused state found", "run_id": run_id}
        return

    state = saved
    state["import_decisions"] = import_decisions

    try:
        # Re-run law availability (picks up newly imported laws)
        yield _step_event(2, "resolve", "running")
        t0 = time.time()
        state = step2c_law_availability(state, db)
        state = step2a_version_selection(state, db)
        state = step2b_concept_search(state, db)
        state = step2d_currency_check(state, db)
        dur = time.time() - t0

        gate_result = step2e_availability_gate(state, db)
        if gate_result:
            yield _step_event(2, "resolve", "done", {"gate_triggered": True}, dur)
            yield gate_result
            return

        yield _step_event(2, "resolve", "done", {"gate_triggered": False}, dur)

        # Continue with Steps 3-5 (same as run_pipeline_v2)
        yield _step_event(3, "retrieve", "running")
        t0 = time.time()
        state = step3_retrieve_per_issue(state, db)
        dur = time.time() - t0
        total_articles = sum(len(v) for v in state.get("issue_articles", {}).values())
        yield _step_event(3, "retrieve", "done", {"total_articles": total_articles}, dur)

        if state.get("complexity") != "SIMPLE":
            yield _step_event(4, "reasoning", "running")
            t0 = time.time()
            state = step4_legal_reasoning(state, db)
            dur = time.time() - t0
            rl_rap = state.get("rl_rap_output")
            yield _step_event(4, "reasoning", "done", {
                "success": rl_rap is not None,
            }, dur)

        yield _step_event(5, "answer", "running")
        t0 = time.time()
        for event in step5_answer_generation(state, db):
            yield event
        state = step5b_citation_validation(state, db)
        confidence = _derive_confidence(state)
        state["confidence"] = confidence
        yield _step_event(5, "answer", "done", {"confidence": confidence}, time.time() - t0)

        complete_run(db, run_id, "ok", confidence, state.get("flags"))
        db.commit()

        yield {
            "type": "done",
            "run_id": run_id,
            "content": state.get("answer", ""),
            "structured": state.get("answer_structured"),
            "mode": state.get("output_mode", "qa"),
            "confidence": confidence,
            "flags": state.get("flags", []),
            "reasoning": _build_v2_reasoning_panel(state),
        }

    except Exception as e:
        logger.exception(f"Pipeline V2 resume error in run {run_id}")
        try:
            complete_run(db, run_id, "error", None, [str(e)])
            db.commit()
        except Exception:
            pass
        yield {"type": "error", "error": str(e), "run_id": run_id}


def _derive_confidence(state: dict) -> str:
    """Derive final confidence from all available signals."""
    rl_rap = state.get("rl_rap_output")
    flags = state.get("flags", [])

    # Start with Claude's confidence or MEDIUM
    confidence = state.get("claude_confidence", "MEDIUM")

    # Downgrade for specific conditions
    if "rl_rap_parse_failed" in flags:
        confidence = "LOW"
    if "rl_rap_truncated" in " ".join(flags):
        confidence = "LOW"
    if "majority_phantom_citations" in flags:
        confidence = "LOW"
    if "low_relevance_signal" in flags:
        if confidence == "HIGH":
            confidence = "MEDIUM"

    # Check RL-RAP certainty levels
    if rl_rap:
        certainties = [i.get("certainty_level", "UNKNOWN") for i in rl_rap.get("issues", [])]
        if "UNCERTAIN" in certainties:
            confidence = "LOW"
        elif all(c == "CERTAIN" for c in certainties) and confidence != "LOW":
            confidence = "HIGH"

    return confidence


def _build_v2_reasoning_panel(state: dict) -> dict:
    """Build the reasoning panel for the frontend."""
    panel = {
        "pipeline_version": "v2",
        "step1_classification": {
            "question_type": state.get("question_type"),
            "legal_domain": state.get("legal_domain"),
            "legal_topic": state.get("legal_topic"),
            "entity_types": state.get("entity_types", []),
            "output_mode": state.get("output_mode"),
            "core_issue": state.get("core_issue"),
            "complexity": state.get("complexity"),
        },
        "step2_resolve": {
            "fact_version_map": state.get("fact_version_map", {}),
            "concept_candidates_per_issue": {
                k: len(v) for k, v in state.get("concept_candidates", {}).items()
            },
            "coverage_status": state.get("coverage_status", {}),
        },
        "step3_retrieve": {
            "per_issue": {
                k: {"count": len(v), "sources": list(set(a.get("source", "?") for a in v))}
                for k, v in state.get("issue_articles", {}).items()
            },
            "total_articles": sum(len(v) for v in state.get("issue_articles", {}).values()),
        },
        "step5_answer": {
            "confidence": state.get("confidence"),
            "phantom_citations": state.get("phantom_citation_count", 0),
            "flags": state.get("flags", []),
        },
    }

    # Step 4: RL-RAP
    rl_rap = state.get("rl_rap_output")
    if rl_rap:
        panel["step4_reasoning"] = {
            "issues_analyzed": len(rl_rap.get("issues", [])),
            "certainty_levels": {
                i["issue_id"]: i.get("certainty_level", "UNKNOWN")
                for i in rl_rap.get("issues", [])
            },
            "operative_articles": state.get("operative_articles", []),
            "governing_norms": {
                i["issue_id"]: i.get("governing_norm_status", {}).get("status", "UNKNOWN")
                for i in rl_rap.get("issues", [])
            },
        }

    return panel
